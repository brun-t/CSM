200Arena_ptr arena_alloc(Arena *arena, size_t size) {
201 if (arena == NULL || size == 0)
202 return (Arena_ptr){.size = 0, .block = NULL};
203
204 if (arena->actual_size + size > arena->capacity)
205 return (Arena_ptr){.size = 0, .block = NULL};
206
207 uint8_t *block = &arena->block[arena->actual_size];
208 arena->actual_size += size;
209
210 return (Arena_ptr){.size = size, .block = block};
211}

253Dyn_ptr *stack_new_ptr(Ptr_stack *stack, void *data, size_t dataSize) {
254 size_t growth = dataSize * 0.9;
255 if (growth < 1024) { // minimum of 1KB growth.
256 growth = 1024;
257 }
258 if (stack->length >= stack->capacity && !arena_realloc(stack->arena, growth)) {
259 return NULL;
260 }
261
262 Arena_ptr arena_ptr = arena_alloc(stack->arena, sizeof(Dyn_ptr));
263 if (arena_ptr.block == NULL) {
264 return NULL;
265 }
266
267 Dyn_ptr *dyn_ptr = (Dyn_ptr *)arena_ptr.block;
268 dyn_ptr->ptr = NULL;

271 stack->length++;
272
273 dyn_ptr = &stack->ptr_list[stack->length - 1];
274 dyn_ptr->size = dataSize;

277 if (dyn_ptr->ptr == NULL)

279 return dyn_ptr;
